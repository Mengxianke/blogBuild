事件中心强持有监听器 ​

ts
class EventEmiiter {
  events = new Map(); // key: eventId. value: [...] cbs;
  
  // 订阅一个topic, 事件名称。一个监听器
  on(eventName, listener) {
    if (!this.events.has(eventName)) {
      this.events.set(eventName, []);
    }
    let listeners = this.events.get(eventName);
    listeners.push(listener);
  }

  // 注销掉事件名称下的某一个listener
  off(eventName, listener) {
    if (this.events.has(eventName)) {
      let listeners = this.events.get(eventName);
      listeners = listeners.filter(l => l !== listener);
    }
  }

  emit(eventName, arg) {
    if (this.events.has(eventName)) {
      this.events.get(eventName).forEach(listener => listener(arg));
    }
  }

  once(eventName, listener) {
    const fn = () => {
      listener();
      this.off(eventName);
    }
    this.on(eventName, fn);
  } 
}

事件中心弱持有监听器 ​

ts
class EventCenter {
    private events: Map<string, Set<WeakRef<Function>>> = new Map();

    // 通过Set<WeakRef<Function>>合集，找到listener
    getWeakListenerFromHandler(listeners: Set<WeakRef<Function>>, target: Function) {
        for (let listener of listeners) {
            const key = listener.deref();
            if (key && key === target) {
                return listener;
            }
        }
        return null;
    }
    // 监听
    on(eventName, listener) {
        if (!this.events.has(eventName)) {
            this.events.set(eventName, new Set());
        }
        // 构造弱引用
        const weakRefListener = new WeakRef(listener);
        this.events.get(eventName).add(weakRefListener);
        // 创建一个终结，传入listener对应的弱引用
        const fr = new FinalizationRegistry((weakFunc: WeakRef<Function>) => {
            this.events.forEach((listeners, eventName) => {
                listeners.delete(weakFunc);
            });
        });
        // 对listener注册一个终结， 传入listener对应的弱引用
        fr.register(listener, weakRefListener);
    }

    // 取消监听，寻找listener与weakRef<>之间的关系
    off(eventName, listener) {
        const listeners = this.events.get(eventName);
        if (!listeners) {
            return;
        }
        const key = this.getWeakListenerFromHandler(listeners, listener);
        if (key) {
            listeners.delete(key);
        }
    }

}

interface Observer {
  // 观察者可以执行的行为

// 具体观察者类
class ConcreteObserver implements Observer {
  
  update() {

  }

}


class Subject {

  observerList: Array<Observer> = [];

  addObserver(observer) {
    this.observerList.push(observer);
  }

  removeObserver(observer) {
    let index = this.observerList.findIndex(o => o === observer);
    if (index > 0) {
      this.observerList.splice(index, 1);
    }
  }
  
  notify() {
    this.observerList.forEach(o => o.update());
  }

}

两者区别 ​

两者都可以是一对多的依赖关系。都是一种消息传递的机制。当一个对象中的状态发生改变时，其所有的依赖者都会收到通知并处理消息。     
观察者模式中, 被观察者(Subject)维护观察者列表。被观察者需要主动调用被观察者的方法，主动建立联系。两者之间有一定的耦合程度。被观察者能够感知到观察者的存在。            
发布订阅模式中, 发布者和订阅者解耦，由事件中心维护关系。 发布者只需要关系自身的发布逻辑， 而不会直接与订阅者直接通信。提供了一种松耦合机制。
