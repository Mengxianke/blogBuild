节流函数 ​

js
// n 秒后在执行该事件，若在 n 秒内被重复触发，则重新计时
function debounce(fn, delay) {
    let timer = null;
    return function(...args) {
        if (timer) {
            clearTimeout(timer); 
        }
        timer = setTimeout(function() {
            fn.apply(this, args);
        }, delay)
    }
}

function deepClone(target, map = new Map()) {
  // 递归出口
  if (typeof target !== 'object') {
      return target;
  }
  if (map.get(target)) {
      return map.get(target);
  }
  let res = {};
  // 设置target, 与其clone的关联关系, 防止循环引用，无限递归, 
  // 对象某个属性为其自身。 先保存该关系。
  map.set(target, res);
  // 遍历
  for (const key in target) {
      if (target.hasOwnProperty(key)) {
          res[key] = deepClone(target[key], map);
      }
  }
  return res;
}

处理更多的类型

js
function deepClone(target, map = new WeakMap()) {
  // 处理null or undefined
  if (target == null || target == undefined) return target;
  if (target instanceof Date) return new Date(obj);
  if (target instanceof RegExp) return new RegExp(obj);
  // 递归出口
  if (typeof target !== 'object') return target;
  if (map.get(target)) return map.get(target);
  let res = {};
  for (const key in target) {
    if (target.hasOwnProperty(key)) {
        res[key] = deepClone(target[key], map);
    }
  }
  // 访问全部子树，返回当前node值
  return res;
}

// left 是构造函数构造出来的实例, right为某个类
// a.isInstaceOf(A)  // a是实例，A是类型
function myInstanceOf(left, right) {
    // 初始化
    left = left.__proto__; // 访问一个实例对应的原型对象
    while(left != null) {
        if (left == right.prototype) { // right为某个类， right.prototype访问其原型对象
            return true;
        }
        left = left.__proto__; // 访问原型的原型
    }
    return false;
}

// arr.reduce((acc, currVal, index, arr) => {}, initalValue)
Array.prototype._reduce = function(fn, initalValue) {
    const arr = this;
    if (this.length === 0) {
        return initalValue;
    }
    // 如果initalValue不存在
    let _initalValue = initalValue;
    let i = 0;
    if (_initalValue === undefined) {
        _initalValue = arr[0]; // 将数组的第一项作为初始值, 简化
        i++;
    }
    let acc = _initalValue;
    for (i; i < arr.length; i++) {
        acc = fn(acc, arr[i], i, arr);
    }
    return acc;
}
