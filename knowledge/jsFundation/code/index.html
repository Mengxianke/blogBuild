js
// n 秒后在执行该事件，若在 n 秒内被重复触发，则重新计时
function debounce(fn, delay) {
    let timer = null;
    return function(...args) {
        if (timer) {
            clearTimeout(timer); 
        }
        timer = setTimeout(function() {
            fn.apply(this, args);
        }, delay)
    }
}

// a.isInstaceOf(A)  // a是实例，A是类型
function myInstanceOf(left, right) {
    // 初始化
    left = left.__proto__; // 访问一个实例对应的原型对象
    while(left != null) {
        if (left == right.prototype) { // right为某个类， right.prototype访问其原型对象
            return true;
        }
        left = left.__proto__; // 访问原型的原型
    }
    return false;
}

// arr.reduce((acc, currVal, index, arr) => {}, initalValue)
Array.prototype._reduce = function(fn, initalValue) {
    const arr = this;
    if (this.length === 0) {
        return initalValue;
    }
    // 如果initalValue不存在
    let _initalValue = initalValue;
    let i = 0;
    if (_initalValue === undefined) {
        _initalValue = arr[0]; // 将数组的第一项作为初始值, 简化
        i++;
    }
    let acc = _initalValue;
    for (i; i < arr.length; i++) {
        acc = fn(acc, arr[i], i, arr);
    }
    return acc;
}

发布订阅模式 ​

ts
class EventEmitter{
    events = {}  // events: key: eventName, value: [....] // value是event的监听器里边存放着callback
    constructor(){
        this.events = {};
    }

    on(eventName, listener){
        //注册事件监听
        // 没有过这个event
        if(!this.events[eventName]){
            this.events[eventName] = []
        }
        if(!this.events[eventName].includes(listener)){
            this.events[eventName].push(listener);
        }
    }

    emit(eventName, arg){
        if(this.events[eventName] && this.events[eventName].length > 0){
            this.events[eventName].forEach(listener => {
                listener(arg) // 给listener推送这个消息
            });
        }
    }

    off(eventName, listener){
        if(this.events[eventName]){
            this.events[eventName] = this.events[eventName].filter(item => item != listener);
        }
    }

    once(eventName, listener){
        // 包装一层
        const fn = ()=>{
            listener();
            this.off(eventName);
        }
        this.on(eventName, fn)
    }
    

}
