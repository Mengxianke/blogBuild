js
const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

class Promise {
    // 状态
    status = PENDING;
    // 解析值
    value = undefined;
    // 失败原因
    reason = undefined;
    // 存放成功的回调
    onResolvedCallbacks = [];
    // 存放失败的回调
    onRejectedCallbacks = [];
    // 存放finally的回调
    onFinallyCallbacks = [];
    // executor = (resolve, reject) => {}
    constructor(executor) {
        // 创建promise为同步方法
        // 解析函数
        let resolve = (value) => {
            // 不可流转特性
            if (this.status === PENDING) {
                this.status = FULFILLED;
                this.value = value;
                this.onResolvedCallbacks.forEach(fn => fn());
                this.onFinallyCallbacks.forEach(fn => fn());
            }
        }
        // 拒绝函数
        let reject = (err) => {
            if (this.status === PENDING) {
                this.status === REJECTED;
                this.reason = err;
                this.onRejectedCallbacks.forEach(fn => fn());
                this.onFinallyCallbacks.forEach(fn => fn());
            }
        }
        try {
            executor(resolve, reject);
        } catch(e) {
            reject(e);
        }
    }
    // then方法, onFulfilled是cb(value), onRejected是cb(reason)
    then(onFulfilled, onRejected) {
        // 已经解析了
        if (this.status === FULFILLED) {
            onFulfilled(this.value);          
        }
        if (this.status === REJECTED) {
            onRejected(this.reason);
        }
        if (this.status === PENDING) {
            // 等待解析，注册回调, 一定推入的是一个返回cb的函数.
            // () => {cb()}
            this.onResolvedCallbacks.push(() => {
                onFulfilled(this.value);
            });
            this.onRejectedCallbacks.push(() => {
                onRejected(this.reason);
            });
        }
    }
    // catch方法, onRejected回调cb
    catch(onRejected) {
        if (this.status === REJECTED) {
            onRejected(this.reason)
        }
        if (this.status === PENDING) {
            this.onRejectedCallbacks.push(() => {
                onRejected(this.reason);
            });
        }
    }
    // finally方法, finally的cb回调
    finally(onFinally) {
        if (this.status === FULFILLED || this.status === REJECTED) {
            onFinally();
        }
        if (this.status === PENDING) {
            // 注册回调
            this.onFinallyCallbacks.push(() => {
                onFinally();
            });
        }
    }
}

Promise A+ 规范 ​

  1. new Promise时，需要传递要给executor执行器, 执行器立刻执行
  2. executor接收两个参数, 分别是resolve和reject
  3. promise 只能从Pending到rejected, 或者从pending到fulfilled
  4. promise的状态一旦确认，就不会再改变
  5. promise都有then方法, then接收两个参数, 分别是promise成功回调onFufilled, promise失败回调onRejected
  6. 如果调用then时候, promise已经成功, 则执行onFufilled, 并将promise的值作为参数传递进去。 如果prommise状态已经失败 ，那么执行OnRejected方法，并将promise失败的原因传递进去。如果Promise的状态是Pending状态，需要将onFulfilled和onRejected函数存放起来， 等待状态确定，再依次将对应的函数执行
  7. then的参数onFulfilled和onRejected可以缺省
  8. promise可以then多次, promise的then方法返回一个promise
  9. 如果then返回的是一个结果, 那么就会把这个结果作为参数，传递给下一个then的回调函数(onFulfilled)
  10. 如果then中抛出了异常, 那么就会把这个异常作为参数，传递给下一个then失败的回调函数(onRejected)
  11. 如果then返回的是一个promise, 那么需要等待这个promise, 这个promise执行完, promise如果成功, 就走下一个then的成功, 如果失败，就走下一个then的失败。

const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

class Promise {
    // 状态
    state;
    // 解析值
    value;
    // 失败原因
    reason;
    // 成功的回调
    fulfilledCbs = [];
    // 失败的回调
    rejectedCbs = [];
    // 构造函数, executor: (resolve, reject) => {}构造器
    constructor(executor) {
        this.state = PENDING; 
        let resolve = (value) => {
            // 不可流转
            if (this.state === PENDING) {
                this.state = FULFILLED;
                this.fulfilledCbs.forEach(cb => cb());
            }
        }
        
        let reject = (reason) => {
            if (this.state === PENDING) {
                this.state = REJECTED;
                this.rejectedCbs.forEach(cb => cb());
            }
        }
        try {
             executor(resolve, reject);
        } catch(e) {
            reject(e);
        }
    }
    // then函数，两个参数
    then(onFulfilled, onRejected) {
        // 给onFulfilled和onRejectd检查, 可以缺省
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled :  (v) => v;
        onRejected = typeof onRejected === 'function' ? onRejected: (reason) => { throw reason };
        // 链式调用, 返回一个promise. then().then().then()
        // 执行onFulfilled和onRejected函数
        let promise2 = new Promise((resolve, reject) => {
            if (this.state === FULFILLED) {
                // 已经解析
                // 因为promise2还没有生成，需要setTimeout
                setTimeout(() => {
                     try {
                        let x = onFulfilled(this.value);
                        // 判断x的类型, 返回的是否是个promise, 通过x.then是否是个函数, 
                        reoslvePromise(promise2, x, resolve, reject);
                    } catch(e) {
                        reject(e);
                    }
                });
            }
            if (this.state === REJECTED) {
                // 已经拒绝
                setTimeout(() => {
                    try {
                        let x = onRejected(this.reason);
                        resolvePromise(promise2, x, resolve, reject);
                    } catch(e) {
                        reject(e);
                    }
                });
            }

            if (this.state === PENDING) {
                // 尚未解析/拒绝, 保存cb
                this.fulfilledCbs.push(() => {
                    setTimeout(() => {
                        try {
                        let x = onFulfilled(this.value);
                        reoslvePromise(promise2, x, resolve, reject);
                        } catch(e) {
                            reject(e);
                        }
                    });
                });
                this.rejectedCbs.push(() => {
                    setTimeout(() => {
                        try {
                            let x = onRejected(this.reason);
                            resolvePromise(promise2, x, resolve, reject);
                        } catch(e) {
                            reject(e); 
                        }
                    });
                })
            }
        });
        return promise2;
    }

    reoslvePromise(promise2, x, resolve, reject) {
        // 判断循环
        if (promise2 === x) reject('Chaining Cycle');
        // 判断x是否是普通类型. 是否是promise, 如果是promise, 解析Promise, 获取得到的结果，传递给下一个.then函数
        if (typeof x === 'object' || typeof x === 'function') {
            try {
                let then = x.then;
                 // 判断then是否是函数
            if (typeof then === 'function') {
                // 通过call调用then函数， 执行then函数
                then.call(x, (y) => {
                    // y是结果, 链式调用，递归调用reolsvePromise
                    resolvePromise(promise2, y, resolve, reject);
                }, (r) => {
                    // 错误
                    reject(r);
                });
            } else {
                // 直接返回
                resolve(x);
                }
            } catch(e) {
                reject(e);
            }
        } else {
            // 是普通类型直接返回
            resolve(x);
        }
    }
}

function promiseAll(promises) {
    const result = [];
    const len = promises.length;
    let count = 0;
    return new Promise((resolve, reject) => {
        promises.forEach((p, index) => {
            p.then((res) => {
                result[index] = res;
                count++;
                if (count === len) {
                    resolve(result);
                }
            }).catch((err) => {
                reject(err);
            });
        });
    });
}

function promiseDefer() {

    deferred.promise = new Promise((resolve, reject) => {
        deferred.resolve = resolve;
        deferred.reject = reject;
    });
    return deferred;
}

// concurrency: 并发数量
export default function pLimit(concurrency) {
    // ...省略对参数校验
    // 维护任务队列， 
    const queue = new Queue();  
    // 标记当前执行的异步任务数量
    let activeCount = 0;
    
    // 自动执行下一个任务
    const resumeNext = () => {
        if (activeCount < concurrency && queue.size > 0) {
            // 执行
            queue.dequeue()();
            activeCount++;
        }
    }

    // 调用下一个任务
    const next = () => {
		activeCount--;
		resumeNext();
	};

    // 执行异步函数, 将resolve传入进来, 内部完成解析, arguments_参数
    const run = async (function_, resolve, arguments_) => {
        const result = (async () => function_(...arguments_))(); // 自执行异步函数
        resolve(result); // 解析结果
        try {
            await result
        } catch {}
        // 执行下一个任务
        next();
    }

    // 将异步函数入列
    const enqueue = (function_, resolve, arguments_) => {
        // 使用run函数对异步函数function_进行包装, 入列一个新函数
        queue.enqueue(run.bind(undefined, function_, resolve, arguments_));
        // 自执行的async函数
        (async () => {
			// This function needs to wait until the next microtask before comparing
			// `activeCount` to `concurrency`, because `activeCount` is updated asynchronously
			// after the `internalResolve` function is dequeued and called. The comparison in the if-statement
			// needs to happen asynchronously as well to get an up-to-date value for `activeCount`.
			await Promise.resolve();

			if (activeCount < concurrency) {
				resumeNext();
			}
		})();
    };


    // 给一个异步函数，返回一个待解析的Promise, 参数function_为一个返回Promise的函数Promise-returning/async function
    const generator = (function_, ...arguments_) => new Promise(resolve => {
        enqueue(function_, resolve, arguments_);
    });


    // 返回生成器
    return generator;
}

为何入列后，不立即执行resumeNext()函数? 必须将执行下一个任务，放到微任务队列中? ​

next()函数中，更新了activeCount变量。这个变量是在等待异步函数解析结果之后，才更新。 如果要获取到最新的，必须要将判断条件进入微任务队列。微任务队列，将微任务逐一取出，执行。更新activeCount变量。

P-Cancelable ​

ts
enum PCancelableStatus {
    PENDING = 'PENDING',
    FULFILLED = 'FULFILLED',
    REJECTED = 'REJECTED',
    CANCELED = 'CANCELED'
}

export class PCancelable<T> {
    // cancelable promise
    private _promise: Promise<T>;
    // 抛出resolve, 外部可以控制解析
    private _resolve?: (value: T) => void;
    // 抛出reject, 外部可以控制拒绝
    private _reject?: Function;
    // 定义promise cancelable状态
    private status: PCancelableStatus = PCancelableStatus.PENDING;
    // 取消的时，执行的一系列回调函数
    private cancelHandlers: Array<Function> = [];
    // executor 执行器, 调用方调用resolve, reject方法, onCacnel(cancelCb), 
    constructor(executor: (
        resolve: (value: T) => void,
        reject: Function,
        onCancel: Function
    ) => void) {
        // 从原始的promise生成一个新的promise
        this._promise = new Promise((resolve, reject) => {
            this._resolve = resolve;
            this._reject = reject;
            const onResolve = (value: T) => {
                // 不可流转特性改写
                if (this.status !== PCancelableStatus.CANCELED && this.status === PCancelableStatus.PENDING) {
                    resolve(value);
                    this.setStatus(PCancelableStatus.FULFILLED);
                }
            };
            const onReject = (reason?: string) => {
                 if (this.status !== PCancelableStatus.CANCELED && this.status === PCancelableStatus.PENDING) {
                    reject(reason);
                    this.setStatus(PCancelableStatus.REJECTED);
                }
            };

            const onCancel = (cancelCb: Function) => {
                // 如果已经不是PENDING状态, 抛出异常
                if (this.status !== PCancelableStatus.PENDING) {
                    new Error(`The onCancel handerl was attached after the promise ${this.status}`);
                }
                this.cancelHandlers.push(cancelCb);
            }
            // 执行执行器
            executor(onResolve, onReject, onCancel);
        });
    }

    private setStatus(status: PCancelableStatus) {
        // 不可流转特性
        if (this.status === PCancelableStatus.PENDING) {
            this.status = status;
        }
    }

    // 注册成功回调
    then<TResult1 = T, TResult2 = never>(onFulfilled: ((value: T) => TResult1 | PromiseLike<TResult1>) | undefined | null, 
    onRejected: ((reason?: string) => TResult2 | PromiseLike<TResult2>) | undefined | null)) {
        return this._promise.then(onFulfilled, onRejected);
    }

    // 注册失败回调
    catch<TResult = never>((onRejected: ((reason?: string) => TResult | PromiseLike<TResult>) | undefined | null)) {
        return this._promise.catch(onRejected);
    }
    // 注册finally回调
    finally(onFinally: () => void) {
        return this._promise.finally(onFinally);
    }
}
