const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

class Promise {
    // 状态
    status = PENDING;
    // 解析值
    value = undefined;
    // 失败原因
    reason = undefined;
    // 存放成功的回调
    onResolvedCallbacks = [];
    // 存放失败的回调
    onRejectedCallbacks = [];
    // 存放finally的回调
    onFinallyCallbacks = [];
    // executor = (resolve, reject) => {}
    constructor(executor) {
        // 创建promise为同步方法
        // 解析函数
        let resolve = (value) => {
            // 不可流转特性
            if (this.status === PENDING) {
                this.status = FULFILLED;
                this.value = value;
                this.onResolvedCallbacks.forEach(fn => fn());
                this.onFinallyCallbacks.forEach(fn => fn());
            }
        }
        // 拒绝函数
        let reject = (err) => {
            if (this.status === PENDING) {
                this.status === REJECTED;
                this.reason = err;
                this.onRejectedCallbacks.forEach(fn => fn());
                this.onFinallyCallbacks.forEach(fn => fn());
            }
        }
        try {
            executor(executor(resolve, reject));
        } catch(e) {
            reject(e);
        }
    }
    // then方法, onFulfilled是cb(value), onRejected是cb(reason)
    then(onFulfilled, onRejected) {
        // 已经解析了
        if (this.status === FULFILLED) {
            onFulfilled(this.value);          
        }
        if (this.status === REJECTED) {
            onRejected(this.reason);
        }
        if (this.status === PENDING) {
            // 等待解析，注册回调, 一定推入的是一个返回cb的函数.
            // () => {cb()}
            this.onResolvedCallbacks.push(() => {
                onFulfilled(this.value);
            });
            this.onRejectedCallbacks.push(() => {
                onRejected(this.reason);
            });
        }
    }
    // catch方法, onRejected回调cb
    catch(onRejected) {
        if (this.status === REJECTED) {
            onRejected(this.reason)
        }
        if (this.status === PENDING) {
            this.onRejectedCallbacks.push(() => {
                onRejected(this.reason);
            });
        }
    }
    // finally方法, finally的cb回调
    finally(onFinally) {
        if (this.status === FULFILLED || this.status === REJECTED) {
            onFinally();
        }
        if (this.status === PENDING) {
            // 注册回调
            this.onFinallyCallbacks.push(() => {
                onFinally();
            });
        }
    }
}

Promise A+ 规范 ​

  1. Promise对象必须有一个then方法，用于注册回调函数，并返回一个新的Promise对象。
  2. then方法接受两个参数：onFulfilled（成功时的回调）和onRejected（失败时的回调）。
  3. then方法必须在Promise状态变化后被调用，且按照特定的顺序执行。 Promise必须支持链式调用，即可以通过多次调用then方法来构建Promise链。

Promise.all ​

function promiseAll(promises) {
    const result = [];
    const len = promises.length;
    let count = 0;
    return new Promise((resolve, reject) => {
        promises.forEach((p, index) => {
            p.then((res) => {
                result[index] = res;
                count++;
                if (count === len) {
                    resolve(result);
                }
            }).catch((err) => {
                reject(err);
            });
        });
    });
}

function PromiseAny(promises) {
    const result = [];
    const len = promises.length;
    let count = 0;
    return new Promise((resolve, reject) => {
        promises.forEach((p, index) => {
            p.then((res) => {
                result[index] = {
                    status: 'fulfilled',
                    value: res
                }
                resolve(result[index]);
            }).catch((err) => {
                result[index] = {
                    status: 'rejected',
                    reason: err
                }
                count++;
                if (count === len) {
                    reject(result);
                }
            });
        });
    });
}

// 借助promise状态的不可流转性
function promiseRace(promises) {
    return new Promise((resolve, reject) => {
        promises.forEach((p, index) => {
            p.then(res => {
                resolve(res);
            }).catch((err) => {
                reject(err);
            });
        });
    });
}

function promiseDefer() {
    const deferred = {};
    deferred.promise = new Promise((resolve, reject) => {
        deferred.resolve = resolve;
        deferred.reject = reject;
    });
    return deferred;
}
